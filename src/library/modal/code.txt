  window.addEventListener(&apos;load&apos;, () =&gt; {
    const close = document.getElementById(&apos;modal-close&apos;);
    const modal = document.getElementById(&apos;modal&apos;);
    const modalOpen = document.querySelectorAll(&apos;[data-modal-opener]&apos;);

    if (!close || !modalOpen || !modal) {
      console.log(&apos;modal init error&apos;, close, modal, modalOpen);
      return;
    }

    modalOpen.forEach((item) =&gt; {
      item.removeAttribute(&apos;data-modal-opener&apos;);
      item.addEventListener(&apos;click&apos;, () =&gt; {
        console.log(&apos;click&apos;);
        modal.setAttribute(&apos;data-open&apos;, false);
        document.body.overflow = &apos;hidden&apos;;
      });
    });

    close.addEventListener(&apos;click&apos;, () =&gt; {
      modal.removeAttribute(&apos;data-open&apos;);
      document.body.overflow = &apos;initial&apos;;
    });
  });
&lt;/script&gt;
